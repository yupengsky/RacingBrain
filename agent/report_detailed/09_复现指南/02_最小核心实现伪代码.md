## 1. 相机节点

```python
class YoloNode(Node):
    def __init__(self):
        super().__init__("yolo_detector")
        self.model = YOLO(model_path)
        self.bridge = CvBridge()
        self.sub = self.create_subscription(Image, "/camera1/image_raw", self.cb, 10)
        self.pub = self.create_publisher(ConeArray, "/yolo/cones", 10)

    def cb(self, image_msg):
        img = self.bridge.imgmsg_to_cv2(image_msg, "bgr8")
        result = self.model(img, conf=0.5, verbose=False)[0]

        out = ConeArray()
        out.header = image_msg.header

        for box in result.boxes:
            x1, y1, x2, y2 = box.xyxy[0].cpu().numpy()
            cls_id = int(box.cls[0])
            cls_name = get_class_name(cls_id)

            cone = Cone()
            cone.center.x = float((x1 + x2) / 2)
            cone.center.y = float((y1 + y2) / 2)
            cone.size.x = float(x2 - x1)
            cone.size.y = float(y2 - y1)
            cone.confidence = float(box.conf[0])
            cone.color = map_class_name_to_color(cls_name)
            out.cones.append(cone)

        self.pub.publish(out)
```

## 2. LiDAR 节点

```cpp
class LidarConeNode : public rclcpp::Node {
public:
    LidarConeNode() : Node("cone_segmentation_node") {
        sub_ = create_subscription<PointCloud2>(
            "/lidar_points", 10,
            std::bind(&LidarConeNode::cb, this, _1)
        );
        pub_ = create_publisher<ThreeDConeArray>("/cone_detection_custom", 10);
    }

    void cb(PointCloud2::ConstSharedPtr msg) {
        pcl::PointCloud<pcl::PointXYZ>::Ptr cloud = from_ros(msg);
        cloud = radius_filter(cloud, 20.0);
        cloud = voxel_grid(cloud, 0.05);

        auto obstacles = remove_ground_by_ransac(cloud);
        auto clusters = euclidean_clusters(obstacles, 0.3, 3, 600);

        ThreeDConeArray out;
        out.header = msg->header;

        for (auto& cluster : clusters) {
            auto [min_pt, max_pt] = get_aabb(cluster);
            double dx = max_pt.x - min_pt.x;
            double dy = max_pt.y - min_pt.y;
            double dz = max_pt.z - min_pt.z;

            if (dz < 0.12 || dz > 0.60) continue;
            if (std::max(dx, dy) > 0.50) continue;
            if (std::max(dx, dy) / std::min(dx, dy) > 3.0) continue;

            ThreeDCone cone;
            cone.center.x = (min_pt.x + max_pt.x) / 2.0;
            cone.center.y = (min_pt.y + max_pt.y) / 2.0;
            cone.center.z = (min_pt.z + max_pt.z) / 2.0;
            cone.size.x = dx;
            cone.size.y = dy;
            cone.size.z = dz;
            out.cones.push_back(cone);
        }

        pub_->publish(out);
    }
};
```

## 3. 融合节点

```cpp
class FusionNode : public rclcpp::Node {
    void cb(ThreeDConeArray lidar, ConeArray camera) {
        Calibration calib = load_calibration();

        std::vector<Detection3D> lidar_boxes = adapt_lidar(lidar);
        std::vector<Detection2D> camera_boxes = adapt_camera(camera);

        std::vector<ProjectedBox> projections;
        for (int i = 0; i < lidar_boxes.size(); ++i) {
            projections.push_back(project_3d_box(lidar_boxes[i], calib, i));
        }

        std::vector<bool> camera_used(camera_boxes.size(), false);
        drd25_msgs::msg::Map out;
        out.header = lidar.header;

        for (auto& pbox : projections) {
            auto& lbox = lidar_boxes[pbox.original_index];

            drd25_msgs::msg::Cone cone;
            cone.x = lbox.center.x;
            cone.y = lbox.center.y;
            cone.color = UNKNOWN;

            int best = -1;
            double best_iou = 0.0;

            if (pbox.valid) {
                for (int j = 0; j < camera_boxes.size(); ++j) {
                    double iou = IoU(pbox.rect, camera_boxes[j].rect);
                    if (iou > best_iou) {
                        best_iou = iou;
                        best = j;
                    }
                }
            }

            if (best >= 0 && best_iou > 0.3) {
                cone.color = camera_boxes[best].color;
                camera_used[best] = true;
            }

            out.track.push_back(cone);
        }

        pub_map.publish(out);
    }
};
```

## 4. SLAM 节点核心

```cpp
class SlamNode : public rclcpp::Node {
    std::vector<GlobalCone> map;
    PJ* proj;
    bool origin_set = false;
    double origin_e, origin_n;
    Matrix4d T_lidar_to_vehicle;

    void sync_cb(Gnss msg, Map perception_map) {
        Pose pose = compute_pose_from_gnss(msg);

        for (auto& c : map) {
            c.matched_this_frame = false;
        }

        for (auto& obs : perception_map.track) {
            if (obs.color == UNKNOWN) continue;

            Vector4d p_lidar(obs.x, obs.y, 0, 1);
            Vector4d p_global = pose.T_vehicle_to_map * T_lidar_to_vehicle * p_lidar;

            update_map(
                Vector2d(p_global.x(), p_global.y()),
                Vector2d(obs.x, obs.y),
                obs.color,
                pose.R2d
            );
        }

        maintain_map(pose.T_vehicle_to_map);
        publish_global_map(msg.header.stamp);
    }

    void update_map(Vector2d z, Vector2d local_xy, int type, Matrix2d vehicle_R) {
        double dist = norm(local_xy);
        Matrix2d R = build_observation_covariance(dist, vehicle_R);

        GlobalCone* best = nullptr;
        double best_d = mahalanobis_thresh;

        for (auto& cone : map) {
            if (cone.type != type) continue;

            Vector2d delta = z - cone.pos;
            if (norm(delta) > max_match_dist) continue;

            Matrix2d S = cone.P + R;
            double d = sqrt(delta.T * inverse(S) * delta);

            if (d < best_d) {
                best_d = d;
                best = &cone;
            }
        }

        if (best) {
            best->P += Identity2 * kf_q_base;
            Matrix2d K = best->P * inverse(best->P + R);
            best->pos = best->pos + K * (z - best->pos);
            best->P = (Identity2 - K) * best->P;
            best->score += dynamic_hit(dist);
            best->score = min(best->score, l_max);
            best->stable = best->score > l_stable;
            best->matched_this_frame = true;
        } else {
            if (map_locked) return;
            if (!in_ackermann_tube(local_xy)) return;

            GlobalCone cone;
            cone.id = next_id++;
            cone.pos = z;
            cone.P = Identity2 * kf_p_init;
            cone.type = type;
            cone.score = dynamic_hit(dist);
            cone.stable = false;
            cone.matched_this_frame = true;
            map.push_back(cone);
        }
    }
};
```

## 5. 地图维护

```cpp
void maintain_map(Matrix4d T_vehicle_to_map) {
    for each cone in map:
        if is_in_fov(cone.pos, T_vehicle_to_map) and !cone.matched_this_frame:
            cone.score += l_miss

        if cone.score <= l_min:
            erase cone
        else:
            cone.stable = cone.score > l_stable
}
```

## 6. 地图发布

```cpp
void publish_global_map(Time stamp) {
    MarkerArray arr;
    arr.markers.push_back(delete_all_marker());

    for (auto& cone : map) {
        if (!cone.stable) continue;

        Marker m;
        m.header.frame_id = "map";
        m.header.stamp = stamp;
        m.id = cone.id;
        m.type = MESH_RESOURCE;
        m.pose.position.x = cone.pos.x();
        m.pose.position.y = cone.pos.y();
        m.color = color_from_type(cone.type);
        arr.markers.push_back(m);
    }

    pub_global.publish(arr);
}
```

## 7. 最小可运行版本与当前工程的差距

最小版本可以暂时省略：

- HSV 颜色检查。
- LiDAR 圆锥拟合。
- CSF 地面分割。
- 单目补漏。
- force match radius。
- 回环锁图。
- 大小黄锥桶 mesh 区分。

但必须保留：

- 时间同步。
- 坐标转换。
- 数据关联。
- Kalman 更新。
- 存在评分。

否则就不再是本工程的核心方案。
